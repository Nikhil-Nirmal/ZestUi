2.  Modal :

A modal is a dialog box/popup window that is displayed on top of a page.

Some examples are shown below :

Hey there.........

×
                
    <div class="modal-container" id="btn-link"> 

        <button id="myBtn" class="btn-outline">Modal</button>

        <div id="myModal" class="modal">

            <div id="modal-content"class="primaryAlert-outline">

                <h1 class="primaryAlertText-outline">Hey there.........</h1>

                <div class="close">×</div>

            </div>

        </div>
        
    </div>
                
            
                

    <script>

        var modal = document.getElementById("myModal");
                        
                
        var btn = document.getElementById("myBtn");
                        
                
        var span = document.getElementsByClassName("close")[0];
                        
                
        btn.onclick = function() 
        {
            modal.style.display = "block";
        }
                        
                
        span.onclick = function() 
        {
            modal.style.display = "none"; 
        }
        window.onclick = function(event) 
        {
            if (event.target == modal) 
            {
                modal.style.display = "none";
            }
        }
    </script>
